#!/bin/bash

# Ruta del archivo
ruta_archivo="/ruta/al/archivo.txt"

# Número de línea a extraer (reemplazar NúmLinea por el número de línea deseado)
numero_linea=NúmLinea

# Extraer las palabras entre corchetes en la línea especificada
palabras=$(sed -n "${numero_linea}p" "$ruta_archivo" | grep -oE '\[([^]]+)\]' | sed 's/\[\|\]//g')

# Imprimir las palabras extraídas
echo "$palabras"
